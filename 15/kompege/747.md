```python
from itertools import *

def f(x):
    P = 43 <= x <= 49
    Q = 44 <= x <= 53
    A = a1 <= x <= a2
    return (A <= P) or Q

oX = [i / 4 for i in range(0, 53*4+1)]
m = []

for a1, a2 in combinations(oX, 2):
    if all(f(x) for x in oX):
        m.append(a2-a1)

print(max(m))
```
