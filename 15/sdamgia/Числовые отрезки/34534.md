#### Решение #2 (Python)
```python
P = range(2, 10 + 1)
Q = range(6, 14 + 1)

def f(x, A):
    global P
    global Q
    return ((x in A) <= (x in P)) or (x in Q)


# [2;6] (6;10] (10; 14]
range1 = range(2, 6 + 1)
range2 = range(6 + 1, 10 + 1)
range3 = range(10 + 1, 14 + 1)


if all( f(x, range1) == True for x in range1 ):
    print( '1 подходит' )

if all( f(x, range2) == True for x in range2 ):
    print( '2 подходит' )

if all( f(x, range3) == True for x in range3 ):
    print( '3 подходит' )

# Получаем, что x ∈ [2; 14]
print(14 - 2) # 12
```
**Ответ:** 12.
