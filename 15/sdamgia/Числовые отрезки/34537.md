```python
minlen = 100000

P = range(10, 15 + 1)
Q = range(10, 20 + 1)
R = range(5, 15 + 1)

for a1 in range(100):
    for a2 in range(100):
        flag = True
        
        for x in range(-100, 100):
            if ((a1 <= x <= a2) <= (x in P)) == ((x in Q) <= (x in R)):
                pass
            else:
                flag = False
                break
        
        if flag:
            minlen = min(minlen, a2 - a1)

print(minlen + 1)
# "за исключением, возможно, конечного числа точек"
```
