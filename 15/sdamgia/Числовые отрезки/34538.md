s = set()

for a1 in range(100):
	for a2 in range(100):
		flag = True
		
		for x in range(-100, 100):
			if (not (a1 <= x <= a2)) <= (not (30 <= x <= 45)):
				if (40 <= x <= 55) <= (a1 <= x <= a2):
					pass
				else:
					flag = False
			else:
				flag = False
		
		if flag:
			s.add(a2 - a1)

print(min(s))
