```python
from itertools import *
chars = '012345678'
wif = ['00','11','22','33','44','55','66','77','88']
count = 0

for i in product(chars,repeat=7):
    lif = 0
    word = ''.join(i)
    if i[0] != '0' and i[0] != '3' and i[0] != '7':
        for l in wif:
            if word.count(l) == 0:
                pass
            else:
               lif = lif + 1
        if lif == 0:
            count = count + 1
print(count)
```
