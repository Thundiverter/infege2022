with open('17_37340.txt') as f:
	s = [int(x) for x in f]
	
	res = []
	
	for i in range(len(s) - 1):
		for j in range(i + 1, len(s)):
			if ((s[i] - s[j]) % 2 == 0) and (s[i] % 31 == 0 or s[j] % 31 == 0):
				res.append(s[i] + s[j])

print(len(res), max(res))
