f = open('24_35482.txt')

a = [i for i in f.readlines()]


# Получаем первую строку с наименьшим количеством букв G

minG = 1000000
rows = []

for j in a:
	if j.count('G') < minG:
		minG = j.count('G')
		rows = [j]
	elif j.count('G') == minG:
		rows.append(j)

r = rows[0]

print(r)

# Считаем количество каждой из букв и находим самую часто используемую
